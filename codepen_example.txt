  position:absolute;

var masterTL = gsap.timeline({paused:true});
var t1 = gsap.timeline();
t1.fromTo($('#square1'), {opacity:0.5, x:200, y:20}, {delay:0, opacity:1, x:200, duration:1.5});
t1.to($('#square1'), {delay:3, opacity:0, x:350, y:20, duration:1.5}, '>');

var t2 = gsap.timeline();
t2.fromTo($('#square2'), {opacity:0.0, x:0, y:20}, {delay:0.0, opacity:1, x:200, duration:1.5});
t2.to($('#square2'), {delay:2, opacity:0, x:350, y:20, duration:1.5}, '>');

var t3 = gsap.timeline();
t3.fromTo($('#square3'), {opacity:0.1, x:0, y:20}, {delay:0.0, opacity:1, x:200, duration:1.5});

masterTL.add(t1);
masterTL.addLabel("box1", '<+0.75');
masterTL.add(t2, ">-1.5");
masterTL.addLabel("box2", '<+1.5');
masterTL.add(t3, ">-1.5");
masterTL.addLabel("box3", '<+1.5');

const btn1 = document.getElementById('playbtn');
btn1.addEventListener('click', function() {masterTL.play(0);});

const btn2 = document.getElementById('prevbtn');
btn2.addEventListener('click', function() {
          masterTL.pause();
          masterTL.seek(masterTL.previousLabel());
});


const btn3 = document.getElementById('nextbtn');
btn3.addEventListener('click', function() { 
          masterTL.pause();
          masterTL.seek(masterTL.nextLabel());
});
